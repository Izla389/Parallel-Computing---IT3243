one to many 

#include <mpi.h> 
#include <stdio.h> 
int main()
{
	int np;
	int pid;
	MPI_Init(NULL, NULL);
	MPI_Comm_size(MPI_COMM_WORLD, &np);
	MPI_Comm_rank(MPI_COMM_WORLD, &pid);
	MPI_Status sta;

	if (pid == 0)
	{
		int num= 3;

		for (int i = 1; i < np; i++) {

			MPI_Send(&num, 1, MPI_INT, i, 50, MPI_COMM_WORLD);

		}

		printf("value sent %d \n", num);


	}
	else
	{
		int recv_num;

		MPI_Recv(&recv_num, 1, MPI_INT, 0, 50, MPI_COMM_WORLD, &sta);
		
			printf("value recieve %d \n", recv_num);
		

	}


	MPI_Finalize();
	return 0;
}

#include <mpi.h> 
#include <stdio.h> 
int main()
{
	int np;
	int pid;
	MPI_Init(NULL, NULL);
	MPI_Comm_size(MPI_COMM_WORLD, &np);
	MPI_Comm_rank(MPI_COMM_WORLD, &pid);
	MPI_Status sta;


	//many to one 
	if (pid == 0)
	{
		int recv_num[8];

		for (int i = 1; i < np; i++) {
			MPI_Recv(&recv_num[i], 1, MPI_INT, i, 50, MPI_COMM_WORLD, &sta);
			printf("value Recieve %d \n", recv_num[i]);
		}

		for (int i = 1; i < np; i++) {
			printf("value From array %d \n", recv_num[i]);
		}

	}

	else
	{
		int num=pid;
		MPI_Send(&num, 1, MPI_INT, 0, 50, MPI_COMM_WORLD);
		printf("value send %d \n", num);

	}



	MPI_Finalize();
	return 0;
}
