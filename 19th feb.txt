 #include <mpi.h> 
#include <stdio.h> 
int main()
{
	int np;
	int pid;
	MPI_Init(NULL, NULL);
	MPI_Comm_size(MPI_COMM_WORLD, &np);
	MPI_Comm_rank(MPI_COMM_WORLD, &pid);

	MPI_Status sta;

	if (pid == 0 ) {
		int num = 5;
		MPI_Send(&num, 1, MPI_INT, 1, 50, MPI_COMM_WORLD);
		printf("PID : %d, sent value: %d\n ", pid, num);
	}
	else if (pid == 1) {
		int recv_num;
		MPI_Recv(&recv_num, 1, MPI_INT, 0, 50, MPI_COMM_WORLD, &sta);
		printf("PID : %d, sent value: %d\n ", pid, recv_num);
	}

	
	MPI_Finalize();
	return 0;
}

#include <mpi.h> 
#include <stdio.h> 
int main()
{
	int np;
	int pid;
	MPI_Init(NULL, NULL);
	MPI_Comm_size(MPI_COMM_WORLD, &np);
	MPI_Comm_rank(MPI_COMM_WORLD, &pid);

	MPI_Status sta;

	if (pid == 0 ) {

		int num[] = {1,2,3,4,5};

		MPI_Send(&num, 5, MPI_INT, 1, 50, MPI_COMM_WORLD);
		printf("My PID is: %d \n I sent ", pid);

		for (int i=0; i <5 ; i ++) {
			printf(" %d",  num[i]);
		}
	}
	else if (pid == 1) {

		int recv_num[5];

		MPI_Recv(&recv_num, 5, MPI_INT, 0, 50, MPI_COMM_WORLD, &sta);
		printf("My PID is: %d \n I recived ", pid);

		for (int i = 0; i < 5; i++) {
			printf(" %d", recv_num[i]);
		}

	}

	
	MPI_Finalize();
	return 0;
}
